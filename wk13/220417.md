bool check(int h) {
    int T[K] = {L}
    for i = N-h : N
        if C[i] < h
           d = h - C[i]
           for j=0:K
               if T[j] > 0 && d>0
                  d--
                  T[j]--
           if d >0    
               return false
    return true;
}

bool check(int h) {
    n_total = 0  // We have K * L citationn
    for i = N-h : N
        if C[i] < h
           d = h - C[i]
           if (d > K)
               return false
           else
               n_total += d
    if n_total <= K * L
        return true;
    else
        return false;
}
